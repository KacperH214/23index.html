        <body>
            <h1>Obliczanie długości boków trójkąta</h1>
            <form>
                Punkt A: <input type="text" id="xa_hebe" placeholder="xa">
                <input type="text" id="ya_hebe" placeholder="ya"><br>
                Punkt B: <input type="text" id="xb_hebe" placeholder="xb">
                <input type="text" id="yb_hebe" placeholder="yb"><br>
                Punkt C: <input type="text" id="xc_hebe" placeholder="xc">
                <input type="text" id="yc_hebe" placeholder="yc"><br>
                <input type="button" value="Oblicz długości boków" onclick="obliczBoki()">
            </form>
            
            <h2>Wyniki:</h2>
            <div id="wyniki"></div>
            
            <script>
                function obliczOdleglosc(x1, y1, x2, y2) {
                    return Math.sqrt(Math.pow(x2 - x1, 2) + Math.pow(y2 - y1, 2));
                }
                
                function obliczBoki() {
                    var xa_hebe = parseFloat(document.getElementById("xa_hebe").value);
                    var ya_hebe = parseFloat(document.getElementById("ya_hebe").value);
                    var xb_hebe = parseFloat(document.getElementById("xb_hebe").value);
                    var yb_hebe = parseFloat(document.getElementById("yb_hebe").value);
                    var xc_hebe = parseFloat(document.getElementById("xc_hebe").value);
                    var yc_hebe = parseFloat(document.getElementById("yc_hebe").value);
                    
                    var ab_hebe = obliczOdleglosc(xa_hebe, ya_hebe, xb_hebe, yb_hebe);
                    var bc_hebe = obliczOdleglosc(xb_hebe, yb_hebe, xc_hebe, yc_hebe);
                    var ac_hebe = obliczOdleglosc(xa_hebe, ya_hebe, xc_hebe, yc_hebe);
                    
                    document.getElementById("wyniki").innerHTML = "Długość AB: " + ab_hebe.toFixed(2) + " cm<br>" +
                                                                    "Długość BC: " + bc_hebe.toFixed(2) + " cm<br>" +
                                                                    "Długość AC: " + ac_hebe.toFixed(2) + " cm";
                }
            </script>
